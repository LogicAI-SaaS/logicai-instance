/**

export interface {{NODE_NAME}}Config {
  // Type de trigger
  triggerType?: 'webhook' | 'schedule' | 'event' | 'polling';

  // Configuration Webhook
  webhookPath?: string;
  webhookMethod?: 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE';
  responseCode?: number;
  responseBody?: string;

  // Configuration Schedule
  cronExpression?: string;
  interval?: number; // en secondes
  timezone?: string;

  // Configuration Event
  eventType?: string;

  // Configuration Polling
  pollInterval?: number; // en millisecondes
  pollUrl?: string;

  // Options
  repeat?: boolean; // Pour répéter le trigger
  maxRuns?: number; // Nombre max d'exécutions
  waitForCompletion?: boolean; // Attendre la fin avant prochain trigger
}

/**
 * État du trigger
 */
interface TriggerState {
  runs: number;
  lastTriggerTime?: Date;
  nextTriggerTime?: Date;
  isActive: boolean;
}

    super(id, name, config);

  /**
   * Point d'entrée principal
   * NOTE: Pour les triggers, execute() démarre le mécanisme de trigger
   */
  async execute(context: ExecutionContext): Promise<NodeExecutionResult> {
    try {
      const triggerType = this.config.triggerType || 'webhook';

      switch (triggerType) {
        case 'webhook':
          return await this.setupWebhook(context);
        case 'schedule':
          return await this.setupSchedule(context);
        case 'event':
          return await this.setupEventListener(context);
        case 'polling':
          return await this.setupPolling(context);
        default:
          throw new Error(`Unknown trigger type: ${triggerType}`);
      }
    } catch (error: any) {
      return {
        success: false,
        error: `Trigger setup failed: ${error.message}`,
      };
    }
  }

  /**
   * Gérer une requête webhook entrante
   * TODO: Implémenter selon les besoins
   */
  private async handleWebhookRequest(req: any, res: any): Promise<NodeExecutionResult> {
    const webhookData = {
      body: req.body,
      headers: req.headers,
      query: req.query,
      method: req.method,
      path: req.path,
    };

    // Répondre immédiatement si configuré
    if (this.config.responseCode) {
      res.status(this.config.responseCode);
      if (this.config.responseBody) {
        res.send(this.config.responseBody);
      } else {
        res.send('OK');
      }
    }

    // Créer le contexte pour les nœuds suivants
    const context: ExecutionContext = {
      $json: webhookData,
      $workflow: {
        id: context.$workflow?.id || 'unknown',
        name: context.$workflow?.name || 'Webhook Workflow',
        active: true,
      },
      $node: {
        id: this.id,
        name: this.name,
        type: this.getType(),
      },
    };

    this.state.runs++;
    this.state.lastTriggerTime = new Date();

    return {
      success: true,
      data: webhookData,
    };
  }

    if (this.config.cronExpression) {
      return await this.setupCronSchedule();
    }

    if (this.config.interval) {
      return await this.setupIntervalSchedule();
    }

    throw new Error('Either cronExpression or interval must be provided');
  }
