import { BaseNode } from '../base/BaseNode';
import { ExecutionContext, NodeExecutionResult } from '../../types';
import axios, { AxiosInstance } from 'axios';

/**
 * Interface de configuration pour {{NODE_NAME}}
 */
export interface {{NODE_NAME}}Config {
  // Authentication
  apiKey?: string;
  token?: string;
  oauthToken?: string;

  // API Configuration
  apiEndpoint?: string;
  apiVersion?: string;
  timeout?: number;

  // Operation
  operation: string; // Ex: 'create', 'read', 'update', 'delete', 'list'
  resource?: string; // Ex: 'users', 'products', 'orders'

  // Request Options
  method?: 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE';
  body?: any;
  params?: Record<string, any>;
  headers?: Record<string, string>;

  // Options
  pagination?: boolean;
  limit?: number;
}

/**
 * Nœud d'intégration pour {{SERVICE_NAME}}
 */
export class {{NODE_NAME}} extends BaseNode {
  private apiClient: AxiosInstance;
  private config: {{NODE_NAME}}Config;

  constructor(id: string, name: string, config: {{NODE_NAME}}Config) {
    super(id, name, config);
    this.config = config;
    this.apiClient = this.initializeAPIClient();
  }

  /**
   * Initialiser le client API
   * TODO: Configurer l'authentification spécifique au service
   */
  private initializeAPIClient(): AxiosInstance {
    const client = axios.create({
      baseURL: this.config.apiEndpoint || 'https://api.{{service}}.com',
      timeout: this.config.timeout || 30000,
      headers: this.buildHeaders(),
    });

    // Interceptor pour les erreurs
    client.interceptors.response.use(
      (response) => response,
      (error) => {
        // TODO: Gérer les erreurs spécifiques (rate limiting, auth, etc.)
        if (error.response?.status === 401) {
          throw new Error('{{SERVICE_NAME}} authentication failed');
        }
        if (error.response?.status === 429) {
          throw new Error('{{SERVICE_NAME}} rate limit exceeded');
        }
        throw error;
      }
    );

    return client;
  }

  /**
   * Construire les headers HTTP avec authentification
   * TODO: Adapter selon le mode d'auth du service
   */
  private buildHeaders(): Record<string, string> {
    const headers: Record<string, string> = {
      'Content-Type': 'application/json',
      'User-Agent': 'LogicAI-N8N/1.0',
      ...this.config.headers,
    };

    // Priorité: OAuth Token > API Key > Basic Auth
    if (this.config.oauthToken) {
      headers['Authorization'] = `Bearer ${this.config.oauthToken}`;
    } else if (this.config.apiKey) {
      // TODO: Vérifier le format attendu (header vs query param)
      headers['X-API-Key'] = this.config.apiKey;
    } else if (this.config.token) {
      headers['Authorization'] = `Token ${this.config.token}`;
    }

    return headers;
  }

  /**
   * Point d'entrée principal - exécute l'opération demandée
   */
  async execute(context: ExecutionContext): Promise<NodeExecutionResult> {
    try {
      const operation = this.config.operation;

      switch (operation) {
        case 'create':
          return await this.create(context);
        case 'read':
          return await this.read(context);
        case 'update':
          return await this.update(context);
        case 'delete':
          return await this.deleteResource(context);
        case 'list':
          return await this.list(context);
        default:
          return await this.customOperation(context, operation);
      }
    } catch (error: any) {
      return {
        success: false,
        error: this.formatErrorMessage(error),
      };
    }
  }

  /**
   * Créer une nouvelle ressource
   * TODO: Adapter selon l'API du service
   */
  private async create(context: ExecutionContext): Promise<NodeExecutionResult> {
    const response = await this.apiClient.post(
      `/${this.config.resource}`,
      this.resolveBody(this.config.body, context)
    );

    return {
      success: true,
      data: {
        ...response.data,
        _operation: 'create',
        _resource: this.config.resource,
      },
    };
  }

  /**
   * Lire une ressource
   * TODO: Adapter selon l'API du service
   */
  private async read(context: ExecutionContext): Promise<NodeExecutionResult> {
    const resourceId = this.getResourceId(context);
    const response = await this.apiClient.get(
      `/${this.config.resource}/${resourceId}`,
      { params: this.config.params }
    );

    return {
      success: true,
      data: response.data,
    };
  }

  /**
   * Mettre à jour une ressource
   * TODO: Adapter selon l'API du service
   */
  private async update(context: ExecutionContext): Promise<NodeExecutionResult> {
    const resourceId = this.getResourceId(context);
    const response = await this.apiClient.put(
      `/${this.config.resource}/${resourceId}`,
      this.resolveBody(this.config.body, context)
    );

    return {
      success: true,
      data: {
        ...response.data,
        _operation: 'update',
      },
    };
  }

  /**
   * Supprimer une ressource
   * TODO: Adapter selon l'API du service
   */
  private async deleteResource(context: ExecutionContext): Promise<NodeExecutionResult> {
    const resourceId = this.getResourceId(context);
    await this.apiClient.delete(`/${this.config.resource}/${resourceId}`);

    return {
      success: true,
      data: {
        _operation: 'delete',
        _resourceId: resourceId,
        message: 'Resource deleted successfully',
      },
    };
  }

  /**
   * Lister les ressources avec pagination
   * TODO: Adapter selon l'API du service
   */
  private async list(context: ExecutionContext): Promise<NodeExecutionResult> {
    const items = [];
    let page = 1;
    let hasMore = true;

    // Récupérer toutes les pages si pagination activée
    while (hasMore && (!this.config.limit || items.length < this.config.limit)) {
      const response = await this.apiClient.get(`/${this.config.resource}`, {
        params: {
          ...this.config.params,
          page,
          per_page: this.config.limit || 100,
        },
      });

      items.push(...response.data.items || response.data);

      // Vérifier s'il y a une page suivante
      hasMore = response.data.hasMore || response.data.length > 0;
      page++;
    }

    return {
      success: true,
      data: {
        items: items.slice(0, this.config.limit || items.length),
        total: items.length,
        _operation: 'list',
      },
    };
  }

  /**
   * Opération personnalisée
   * TODO: Implémenter selon les besoins
   */
  private async customOperation(
    context: ExecutionContext,
    operation: string
  ): Promise<NodeExecutionResult> {
    // Implémentation par défaut: utiliser HTTP method direct
    const method = this.config.method || 'POST';
    const response = await this.apiClient.request({
      method,
      url: this.config.resource,
      data: this.config.body,
      params: this.config.params,
    });

    return {
      success: true,
      data: response.data,
    };
  }

  /**
   * Récupérer l'ID de ressource depuis le context ou la config
   */
  private getResourceId(context: ExecutionContext): string {
    // Essayer depuis $json
    if (context.$json?.id) {
      return String(context.$json.id);
    }
    // Essayer depuis la config
    if (this.config.params?.id) {
      return String(this.config.params.id);
    }
    throw new Error('Resource ID not found in context or config');
  }

  /**
   * Résoudre les variables template dans le body
   */
  private resolveBody(body: any, context: ExecutionContext): any {
    if (typeof body === 'string') {
      return this.resolveVariables(body, context);
    }
    if (Array.isArray(body)) {
      return body.map((item) => this.resolveBody(item, context));
    }
    if (typeof body === 'object' && body !== null) {
      const resolved: any = {};
      for (const [key, value] of Object.entries(body)) {
        resolved[key] = this.resolveBody(value, context);
      }
      return resolved;
    }
    return body;
  }

  /**
   * Résoudre les variables {{ variable }}
   */
  private resolveVariables(text: string, context: ExecutionContext): string {
    return text.replace(/\{\{(\$json|\$workflow|\$node)\.([\w.]+)\}\}/g, (match, type, path) => {
      const obj = type === '$json' ? context.$json : type === '$workflow' ? context.$workflow : context.$node;
      return this.getNestedValue(obj, path) || match;
    });
  }

  /**
   * Formater les messages d'erreur
   */
  private formatErrorMessage(error: any): string {
    if (error.response) {
      const status = error.response.status;
      const data = error.response.data;
      return `${{SERVICE_NAME}} API error ${status}: ${JSON.stringify(data)}`;
    }
    if (error.request) {
      return `${{SERVICE_NAME}} API not responding`;
    }
    return error.message || 'Unknown error';
  }

  /**
   * Type du nœud
   */
  getType(): string {
    return '{{NODE_TYPE}}'; // TODO: Remplacer par le type (ex: 'gmail', 'slack')
  }

  /**
   * Icône du nœud
   */
  getIcon(): string {
    return '{{ICON_NAME}}'; // TODO: Remplacer par l'icône Lucide (ex: 'Mail', 'MessageCircle')
  }
}
