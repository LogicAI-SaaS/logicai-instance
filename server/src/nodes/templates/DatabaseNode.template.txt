  private buildWhereClause(where: Record<string, any> | undefined): string {
    if (!where || Object.keys(where).length === 0) {
      return '';

    const conditions = Object.keys(where).map((key) => `${key} = ?`);
    return ` WHERE ${conditions.join(' AND ')}`;
  }
